136void ArrowMarker::setArrowFromPoints(const MarkerConstSharedPtr & message)
137{
138  last_arrow_set_from_points_ = true;
139 
140  // compute translation & rotation from the two points
141  Ogre::Vector3 point1 = rviz_common::pointMsgToOgre(message->points[0]);
142  Ogre::Vector3 point2 = rviz_common::pointMsgToOgre(message->points[1]);
143 
144  Ogre::Vector3 direction = point2 - point1;
145  float distance = direction.length();
146

149  if (message->scale.z != 0.0) {

151    head_length = std::max<double>(0.0, std::min<double>(length, distance));  // clamp
152  }
153  float shaft_length = distance - head_length;
154 
155  arrow_->set(shaft_length, message->scale.x, head_length, message->scale.y);
156 
157  direction.normalise();
158 
159  // for some reason the arrow goes into the y direction by default
160  Ogre::Quaternion orient = Ogre::Vector3::NEGATIVE_UNIT_Z.getRotationTo(direction);
161 
162  // if scale.x and scale.y are 0, then nothing is shown
163  if (owner_ && (message->scale.x + message->scale.y == 0.0f)) {
164    owner_->setMarkerStatus(
