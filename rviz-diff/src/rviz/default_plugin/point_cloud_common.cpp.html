132Ogre::Vector3 pointFromCloud(const sensor_msgs::PointCloud2ConstPtr& cloud, uint32_t index)
133{
134  int32_t xi = findChannelIndex(cloud, "x");
135  int32_t yi = findChannelIndex(cloud, "y");
136  int32_t zi = findChannelIndex(cloud, "z");
137 
138  const uint32_t xoff = cloud->fields[xi].offset;
139  const uint32_t yoff = cloud->fields[yi].offset;
140  const uint32_t zoff = cloud->fields[zi].offset;
141  const uint8_t type = cloud->fields[xi].datatype;
142  const uint32_t point_step = cloud->point_step;
143  float x = valueFromCloud<float>(cloud, xoff, type, point_step, index);
144  float y = valueFromCloud<float>(cloud, yoff, type, point_step, index);
145  float z = valueFromCloud<float>(cloud, zoff, type, point_step, index);
146  return Ogre::Vector3(x, y, z);
147}

149void PointCloudSelectionHandler::createProperties( const Picked& obj, Property* parent_property )
150{
151  typedef std::set<int> S_int;
152  S_int indices;
153  {
154    S_uint64::const_iterator it = obj.extra_handles.begin();
155    S_uint64::const_iterator end = obj.extra_handles.end();
156    for (; it != end; ++it)
157    {
158      uint64_t handle = *it;
159      indices.insert((handle & 0xffffffff) - 1);
160    }
161  }
162 
163  {
164    S_int::iterator it = indices.begin();
165    S_int::iterator end = indices.end();
166    for (; it != end; ++it)
167    {
168      int index = *it;
169      const sensor_msgs::PointCloud2ConstPtr& message = cloud_info_->message_;
170 
171      IndexAndMessage hash_key( index, message.get() );
172      if( !property_hash_.contains( hash_key ))
173      {
174        Property* cat = new Property( QString( "Point %1 [cloud 0x%2]" ).arg( index ).arg( (uint64_t) message.get() ),
175                                      QVariant(), "", parent_property );
176        property_hash_.insert( hash_key, cat );
177 
178        // First add the position.
179        VectorProperty* pos_prop = new VectorProperty( "Position", cloud_info_->transformed_points_[index].position, "", cat );
180        pos_prop->setReadOnly( true );
181 
182        // Then add all other fields as well.
183        for( size_t field = 0; field < message->fields.size(); ++field )
184        {
185          const sensor_msgs::PointField& f = message->fields[ field ];
186          const std::string& name = f.name;
187 
188          if( name == "x" || name == "y" || name == "z" || name == "X" || name == "Y" || name == "Z" )
189          {
190            continue;
191          }
192          if( name == "rgb" || name == "rgba")
193          {
194            float float_val = valueFromCloud<float>( message, f.offset, f.datatype, message->point_step, index );
195            // Convertion hack because rgb are stored int float (datatype=7) and valueFromCloud can't cast float to uint32_t
196            uint32_t val = *((uint32_t*) &float_val);
197            ColorProperty* prop = new ColorProperty( QString( "%1: %2" ).arg( field ).arg( QString::fromStdString( name )),
198                                                     QColor( (val >> 16) & 0xff, (val >> 8) & 0xff, val & 0xff), "", cat );
199            prop->setReadOnly( true );
200 
201            FloatProperty* aprop = new FloatProperty( QString( "alpha" ), ((val >> 24) / 255.0), "", cat );
202            aprop->setReadOnly( true );
203          }
204          else
205          {
206            float val = valueFromCloud<float>( message, f.offset, f.datatype, message->point_step, index );
207            FloatProperty* prop = new FloatProperty( QString( "%1: %2" ).arg( field ).arg( QString::fromStdString( name )),
208                                                     val, "", cat );
209            prop->setReadOnly( true );
210          }
211        }
212      }
213    }
214  }
