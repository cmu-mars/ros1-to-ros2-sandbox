nn203void Robot::update(const LinkUpdater & updater)
204{
205  for (const auto & link_entry : links_) {
206    RobotLink * link = link_entry.second;
207 
208    link->setToNormalMaterial();
209 
210    Ogre::Vector3 visual_position, collision_position;
211    Ogre::Quaternion visual_orientation, collision_orientation;
212    if (updater.getLinkTransforms(
213        link->getName(),
214        visual_position,
215        visual_orientation,
216        collision_position,
217        collision_orientation))
218    {
219      // Check if visual_orientation, visual_position, collision_orientation,
220      // and collision_position are NaN.
221      if (visual_orientation.isNaN()) {
222        log_error(link, "visual", "orientation");
223        continue;
224      }
225      if (visual_position.isNaN()) {
226        log_error(link, "visual", "position");
227        continue;
228      }
229      if (collision_orientation.isNaN()) {
230        log_error(link, "collision", "orientation");
231        continue;
232      }
233      if (collision_position.isNaN()) {
234        log_error(link, "collision", "position");
235        continue;
236      }
237      link->setTransforms(visual_position, visual_orientation, collision_position,
238        collision_orientation);
239 
240      for (const auto & child_joint_name : link->getChildJointNames()) {
241        RobotJoint * joint = getJoint(child_joint_name);
242        if (joint) {
243          joint->setTransforms(visual_position, visual_orientation);
244        }
245      }
246    } else {
247      link->setToErrorMaterial();
248    }
249  }
